Content usando [Flexbox](https://www.w3.org/TR/css-flexbox-1/)

## Default
```jsx
<Content>
  <span>Content</span>
  <span>Content</span>
  <span>Content</span>
</Content>
```

## Column
```jsx
<Content column>
  <span>Content</span>
  <span>Content</span>
  <span>Content</span>
</Content>
```

## Loading
```jsx
<Content loading>
  <span>Content</span>
  <span>Content</span>
  <span>Content</span>
</Content>
```

## alignItems
```jsx
<Content alignItems="center" column>
  <span>Content</span>
  <span>Content</span>
  <span>Content</span>
</Content>
```

## justifyContent
```jsx
<Content justifyContent="center">
  <span>Content</span>
  <span>Content</span>
  <span>Content</span>
</Content>
```
